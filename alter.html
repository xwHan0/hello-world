ALTER TABLE table-name
command

  • Add-statement: ::= ADD column-name column-type
    • * ::= Select all columns
    • table ::= Table name or table alias name
    • col ::= Column name
  • Delete-statement: ::= DROP COLUMN column-name
    • There are two statements
      • Common: tbl-identifier,...
      • Union: tbl-identifier0 join-statement tbl-identifier1 on union-condition-statement
    • tbl-identifier ::= tbl-name [ alias-name]
  • Edit-statement: ::= ALTER COLUMN column-name type
